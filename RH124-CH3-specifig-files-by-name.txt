$cd - # cambiar al directorio previo
$ cd ..  # se mueve un nivel hacia arriba en el directorio padre

$ pwd # muestra la ubicacion actual (ruta absoluta del directorio actual)

$ touch archivo.txt # touch crea archivos vacios, ene ste caso crea el archivo vacio  "archivo.txt"
